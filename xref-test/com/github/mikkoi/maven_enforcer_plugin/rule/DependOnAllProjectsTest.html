19      @Test
20      void testProjectsContains() {
21          final List<MavenProject> reactorProjects = new ArrayList<>();
22          for (String s : Arrays.asList("mikkoi:proj-a:pom", "mikkoi:proj-b:jar",
23              "mikkoi:proj-c:war")) {
24              Model model = new Model();
25              model.setGroupId(Arrays.asList(s.split(":")).get(0));
26              model.setArtifactId(Arrays.asList(s.split(":")).get(1));
27              model.setPackaging(Arrays.asList(s.split(":")).get(2));
28              reactorProjects.add(new MavenProject(model));
29          }
30          assertThat(
31              DependOnAllProjects.projectsContains(reactorProjects, "mikkoi:proj-a:pom")).isTrue();
32          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "mikkoi:proj-a")).isTrue();
33          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "proj-a")).isTrue();
34          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "*")).isFalse();
35          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "mikkoi:*")).isFalse();
36          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "mikkoi:*:pom")).isFalse();
37          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "*:*:pom")).isFalse();
38          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "proj-*")).isFalse();
39          assertThat(DependOnAllProjects.projectsContains(reactorProjects, "proj-d")).isFalse();
40          assertThat(
41              DependOnAllProjects.projectsContains(reactorProjects, "mikkoi:proj-a:war")).isFalse();
42      }

44      @Test
45      void testIsProjectIncluded() {
46          final MavenProject project = new MavenProject();
47  
48          List<String> includes = new ArrayList<>();
49          List<String> excludes = new ArrayList<>();
50  
51          includes.add("*");
52          project.setGroupId("com.github.mikkoi");
53          project.setArtifactId("test-artifact");
54          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isTrue();
55  
56          includes.add("com.github.mikkoi:*");
57          excludes.add("com.github.mikkoi:test-artifact");
58  
59          project.setGroupId("com.github.mikkoi");
60          project.setArtifactId("test-artifact");
61          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isFalse();
62  
63          project.setArtifactId("test-artifact-2");
64          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isTrue();
65  
66          excludes.add("com.gitlab.other:other-artifact");
67          project.setArtifactId("other-artifact");
68          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isTrue();
69  
70          project.setGroupId("com.gitlab.other");
71          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isFalse();
72  
73          excludes.add("com.gitlab.second:*-other-artifact");
74          project.setGroupId("com.gitlab.second");
75          project.setArtifactId("diff-other-artifact");
76          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isFalse();
77  
78          project.setArtifactId("artifact-something");
79          assertThat(DependOnAllProjects.isProjectIncluded(includes, excludes, project)).isTrue();
80      }

82      @Test
83      void testConvertStringForMatching() {
84          String name1 = DependOnAllProjects.convertStringForMatching("apache");
85          assertThat(name1).isEqualTo(".*:apache:.*");
86  
87          String name2 = DependOnAllProjects.convertStringForMatching("org.apache.maven:core:jar");
88          assertThat(name2).isEqualTo("org\\.apache\\.maven:core:jar");
89  
90          String name3 = DependOnAllProjects.convertStringForMatching("org.apache.maven:core");
91          assertThat(name3).isEqualTo("org\\.apache\\.maven:core:.*");
92  
93          String name4 = DependOnAllProjects.convertStringForMatching("org.apache.*:*");
94          assertThat(name4).isEqualTo("org\\.apache\\..*:.*:.*");
95  
96          String name5 =
97              DependOnAllProjects.convertStringForMatching("com.github.mikkoi:test-artifact");
98          assertThat(name5).isEqualTo("com\\.github\\.mikkoi:test-artifact:.*");
