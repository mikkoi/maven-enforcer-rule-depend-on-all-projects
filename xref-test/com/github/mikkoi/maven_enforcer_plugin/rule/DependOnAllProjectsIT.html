81      private Dependency getRootProjectAsDependency(MavenExecutionResult result) {
82          final Dependency rootProject = new Dependency();
83          rootProject.setType(result.getMavenProjectResult().getModel().getPackaging());
84          rootProject.setGroupId(result.getMavenProjectResult().getModel().getGroupId());
85          rootProject.setArtifactId(result.getMavenProjectResult().getModel().getArtifactId());
86          rootProject.setVersion(result.getMavenProjectResult().getModel().getVersion());
87          return rootProject;
